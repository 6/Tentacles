### install

    git clone https://github.com/6/sinatra-backbone-boilerplate.git your_app
    cd your_app
    git remote rm origin
    git remote add upstream https://github.com/6/sinatra-backbone-boilerplate.git
    bundle install

### run

    foreman start

### pulling in updates

    git fetch upstream
    git merge upstream/master
